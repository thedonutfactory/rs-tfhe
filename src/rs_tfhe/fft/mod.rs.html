1//! FFT Processor Module for TFHE
2//!
3//! This module provides abstracted FFT operations for negacyclic polynomial
4//! multiplication in the ring R[X]/(X^N+1), which is fundamental to TFHE.
5//!
6//! # Architecture
7//!
8//! The module uses a trait-based design with platform-specific implementations:
9//!
10//! - **x86_64**: `SpqliosFFT` - Hand-optimized AVX/FMA assembly (~30ms/gate)
11//! - **ARM64/Other**: `RustFFTProcessor` - Pure Rust implementation (~100ms/gate)
12//!
13//! Both implementations are mathematically equivalent and pass identical test suites.
14//!
15//! # Usage
16//!
17//! ```ignore
18//! use crate::fft::{DefaultFFTProcessor, FFTProcessor};
19//!
20//! let mut processor = DefaultFFTProcessor::new(1024);
21//! let freq = processor.ifft_1024(&time_domain);
22//! let result = processor.fft_1024(&freq);
23//! ```
24//!
25//! # Algorithm
26//!
27//! The negacyclic FFT embeds an N-point negacyclic problem into a 2N-point
28//! cyclic FFT, extracting only the odd frequency indices which correspond
29//! to the primitive 2N-th roots of unity needed for polynomial multiplication
30//! modulo X^N+1.

76/// FFT Processor trait for negacyclic polynomial multiplication in R[X]/(X^N+1)
77///
78/// All implementations must provide mathematically equivalent operations
79/// for TFHE's core polynomial arithmetic.
80pub trait FFTProcessor {
81  /// Create a new FFT processor for polynomials of size n
82  fn new(n: usize) -> Self;
83
84  /// Generic forward FFT for any power-of-2 size N
85  /// Input: N torus32 values representing polynomial coefficients
86  /// Output: N f64 values (N/2 complex stored as [re_0..re_N/2-1, im_0..im_N/2-1])
87  fn ifft<const N: usize>(&mut self, input: &[Torus; N]) -> [f64; N];
88
89  /// Generic inverse FFT for any power-of-2 size N
90  /// Input: N f64 values (N/2 complex stored as [re_0..re_N/2-1, im_0..im_N/2-1])
91  /// Output: N torus32 values representing polynomial coefficients
92  fn fft<const N: usize>(&mut self, input: &[f64; N]) -> [Torus; N];
93
94  /// Generic negacyclic polynomial multiplication for any power-of-2 size N
95  /// Computes: a(X) * b(X) mod (X^N+1)
96  fn poly_mul<const N: usize>(&mut self, a: &[Torus; N], b: &[Torus; N]) -> [Torus; N];
97
98  /// Generic batch IFFT for any power-of-2 size N
99  fn batch_ifft<const N: usize>(&mut self, inputs: &[[Torus; N]]) -> Vec<[f64; N]> {
100    inputs.iter().map(|input| self.ifft::<N>(input)).collect()
101  }
102
103  /// Generic batch FFT for any power-of-2 size N
104  fn batch_fft<const N: usize>(&mut self, inputs: &[[f64; N]]) -> Vec<[Torus; N]> {
105    inputs.iter().map(|input| self.fft::<N>(input)).collect()
106  }
107}

240  fn poly_mul<const N: usize>(a: &[Torus; N], b: &[Torus; N]) -> [Torus; N] {
241    let n = a.len();
242    let mut res: Vec<Torus> = vec![0; n];
243
244    for i in 0..n {
245      for j in 0..n {
246        if i + j < n {
247          res[i + j] = res[i + j].wrapping_add(a[i].wrapping_mul(b[j]));
248        } else {
249          res[i + j - n] = res[i + j - n].wrapping_sub(a[i].wrapping_mul(b[j]));
250        }
251      }
252    }
253
254    res.try_into().unwrap()
255  }
