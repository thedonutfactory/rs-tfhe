1/// TFHE Security Parameter Selection
2///
3/// This library supports multiple security levels to allow users to choose
4/// the right balance between performance and security for their use case.
5///
6/// # Available Security Levels
7///
8/// - **80-bit**: Fast performance, suitable for development/testing
9/// - **110-bit**: Balanced performance and security (original TFHE reference)
10/// - **128-bit**: High security, quantum-resistant (default)
11/// - **Uint1-Uint8**: Specialized parameters for different message moduli
12///
13/// # Security Parameters Explained
14///
15/// The security level is determined by several cryptographic parameters:
16/// - `N`: LWE dimension (higher = more secure, slower)
17/// - `ALPHA`: Noise standard deviation (smaller = often more secure with proper dimension)
18/// - `L`: Gadget decomposition levels (more = more secure, slower)
19/// - `BGBIT`: Decomposition base bits (smaller = more levels, more secure, slower)
20///
21/// # Usage Example
22///
23/// ```rust
24/// use rs_tfhe::params::SECURITY_128_BIT;
25///
26/// // Use 128-bit security (default)
27/// let params = SECURITY_128_BIT;
28/// ```
29///
30/// # LUT Bootstrapping Parameters
31///
32/// ```rust
33/// #[cfg(feature = "lut-bootstrap")]
34/// use rs_tfhe::params::SECURITY_UINT5;
35///
36/// #[cfg(feature = "lut-bootstrap")]
37/// // Use Uint5 parameters for complex arithmetic
38/// let params = SECURITY_UINT5;
39/// ```
40pub type Torus = u32;
41pub type HalfTorus = i32;
42pub type IntTorus = i64;
43
44pub const TORUS_SIZE: usize = std::mem::size_of::<Torus>() * 8;
45pub const ZERO_TORUS: Torus = 0;

51/// Security parameter set containing all TFHE parameters
52#[derive(Debug, Clone, Copy)]
53pub struct SecurityParams {
54  pub security_bits: usize,
55  pub description: &'static str,
56  pub tlwe_lv0: TlweParams,
57  pub tlwe_lv1: TlweParams,
58  pub trlwe_lv1: TrlweParams,
59  pub trgsw_lv1: TrgswParams,
60}
61
62#[derive(Debug, Clone, Copy)]
63pub struct TlweParams {
64  pub n: usize,
65  pub alpha: f64,
66}
67
68#[derive(Debug, Clone, Copy)]
69pub struct TrlweParams {
70  pub n: usize,
71  pub alpha: f64,
72}
73
74#[derive(Debug, Clone, Copy)]
75pub struct TrgswParams {
76  pub n: usize,
77  pub nbit: usize,
78  pub bgbit: u32,
79  pub bg: u32,
80  pub l: usize,
81  pub basebit: usize,
82  pub iks_t: usize,
83  pub alpha: f64,
84}

413/// Get a description of the current security level
414pub fn security_info(params: SecurityParams) -> String {
415  format!(
416    "Security level: {} bits ({})",
417    params.security_bits, params.description
418  )
