1//! Lookup Table (LUT) support for programmable bootstrapping
2//!
3//! This module provides functionality for creating and using lookup tables
4//! in programmable bootstrapping operations. LUT bootstrapping allows
5//! evaluating arbitrary functions on encrypted data during the bootstrapping
6//! process, combining noise refreshing with function evaluation.
7//!
8//! # Key Concepts
9//!
10//! - **Lookup Table**: A TRLWE ciphertext that encodes a function for evaluation
11//! - **Programmable Bootstrapping**: Apply arbitrary functions during bootstrapping
12//! - **Message Encoding**: Support for different message moduli (binary, multi-bit)
13//! - **Function Evaluation**: Evaluate f(x) on encrypted x during bootstrapping
14//!
15//! # Example
16//!
17//! ```rust
18//! use rs_tfhe::lut::{LookupTable, Generator};
19//! use rs_tfhe::bootstrap::lut::LutBootstrap;
20//! use rs_tfhe::key;
21//! use rs_tfhe::utils::Ciphertext;
22//! use rs_tfhe::params;
23//!
24//! // Create a generator for binary messages
25//! let generator = Generator::new(2);
26//!
27//! // Define a function (e.g., NOT)
28//! let not_func = |x: usize| 1 - x;
29//!
30//! // Generate lookup table
31//! let lut = generator.generate_lookup_table(not_func);
32//!
33//! // Use in programmable bootstrapping
34//! let bootstrap = LutBootstrap::new();
35//! let secret_key = key::SecretKey::new();
36//! let cloud_key = key::CloudKey::new(&secret_key);
37//! let ciphertext = Ciphertext::encrypt_lwe_message(1, 2, params::tlwe_lv0::ALPHA, &secret_key.key_lv0);
38//! let result = bootstrap.bootstrap_lut(&ciphertext, &lut, &cloud_key);
39//! ```
40

42pub mod generator;
43pub mod lookup_table;
