9/// Trait for bootstrapping strategies
10///
11/// Bootstrapping is the core operation in TFHE that refreshes a noisy ciphertext
12/// by homomorphically evaluating the decryption function. Different strategies
13/// can optimize for different tradeoffs (speed, accuracy, memory, hardware).
14///
15/// # Core Operation
16/// Bootstrap takes a noisy LWE ciphertext at level 0 and returns a refreshed
17/// ciphertext with reduced noise, enabling unbounded homomorphic operations.
18///
19/// # Typical Flow
20/// 1. Blind rotation: Homomorphically evaluate a test polynomial
21/// 2. Sample extraction: Convert RLWE to LWE
22/// 3. Key switching: Convert back to original key
23pub trait Bootstrap: Send + Sync {
24  /// Perform a full bootstrap with key switching
25  ///
26  /// Takes a potentially noisy ciphertext and returns a refreshed one
27  /// under the original encryption key.
28  fn bootstrap(&self, ctxt: &Ciphertext, cloud_key: &CloudKey) -> Ciphertext;
29
30  /// Perform bootstrap without key switching
31  ///
32  /// Returns a ciphertext under the expanded RLWE key (level 1).
33  /// Useful when chaining multiple operations before final key switch.
34  fn bootstrap_without_key_switch(&self, ctxt: &Ciphertext, cloud_key: &CloudKey) -> Ciphertext;
35
36  /// Get the name of this bootstrap strategy
37  fn name(&self) -> &str;
38}
39
40/// Get the default bootstrap strategy
41pub fn default_bootstrap() -> Box<dyn Bootstrap> {
42  Box::new(vanilla::VanillaBootstrap::new())
43}
