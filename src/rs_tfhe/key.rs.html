26impl SecretKey {
27  pub fn new() -> Self {
28    let mut rng = rand::thread_rng();
29    let mut key = SecretKey {
30      key_lv0: [ZERO_TORUS; params::tlwe_lv0::N],
31      key_lv1: [ZERO_TORUS; params::tlwe_lv1::N],
32    };
33    key
34      .key_lv0
35      .iter_mut()
36      .for_each(|e| *e = rng.gen::<bool>() as Torus);
37    key
38      .key_lv1
39      .iter_mut()
40      .for_each(|e| *e = rng.gen::<bool>() as Torus);
41    key
42  }
43}

72pub fn gen_decomposition_offset() -> Torus {
73  let mut offset: Torus = 0;
74
75  for i in 0..(params::trgsw_lv1::L as Torus) {
76    offset = offset.wrapping_add(
77      params::trgsw_lv1::BG / 2
78        * (1 << (TORUS_SIZE - (i + 1) as usize * params::trgsw_lv1::BGBIT as usize)),
79    );
80  }
81
82  offset
83}

85pub fn gen_testvec() -> trlwe::TRLWELv1 {
86  let mut testvec = trlwe::TRLWELv1::new();
87  let b_torus = utils::f64_to_torus(0.125);
88  for i in 0..params::trgsw_lv1::N {
89    testvec.a[i] = 0;
90    testvec.b[i] = b_torus;
91  }
92
93  testvec

96pub fn gen_key_switching_key(secret_key: &SecretKey) -> KeySwitchingKey {
97  const BASEBIT: usize = params::trgsw_lv1::BASEBIT;
98  const IKS_T: usize = params::trgsw_lv1::IKS_T;
99  let mut res = vec![tlwe::TLWELv0::new(); TRGSWLV1_BASE * TRGSWLV1_IKS_T * TRGSWLV1_N];
100
101  for i in 0..params::trgsw_lv1::N {
102    for j in 0..IKS_T {
103      for k in 0..TRGSWLV1_BASE {
104        if k == 0 {
105          continue;
106        }
107        let p =
108          ((k as u32 * secret_key.key_lv1[i] as u32) as f64) / ((1 << ((j + 1) * BASEBIT)) as f64);
109        let idx = (TRGSWLV1_BASE * TRGSWLV1_IKS_T * i) + (TRGSWLV1_BASE * j) + k;
110        res[idx] = tlwe::TLWELv0::encrypt_f64(p, params::KSK_ALPHA, &secret_key.key_lv0);
111      }
112    }
113  }
114
115  res
116}

122pub fn gen_bootstrapping_key_with_railgun<R: crate::parallel::Railgun>(
123  secret_key: &SecretKey,
124  railgun: &R,
125) -> BootstrappingKey {
126  // OPTIMIZATION: Parallel bootstrap key generation
127  // The bootstrap key has 500-700 independent TRGSW encryptions (one per LWE coefficient)
128  // Each encryption is completely independent and can be parallelized
129  // Expected speedup: 4-8x on multi-core systems
130  //
131  // Each worker thread uses its own thread-local FFT_PLAN for efficient caching
132  //
133  // NOTE: Threads need larger stack (8MB) due to deep call chains in TRGSW encryption
134  let config = crate::parallel::ParallelConfig {
135    stack_size: Some(8 * 1024 * 1024), // 8MB stack per thread
136    num_threads: None,
137  };
138
139  railgun.with_config(config, || {
140    railgun.par_map(&secret_key.key_lv0, |&kval| {
141      FFT_PLAN.with(|plan| {
142        let p = &mut plan.borrow_mut();
143        trgsw::TRGSWLv1FFT::new(
144          &trgsw::TRGSWLv1::encrypt_torus(kval, params::BSK_ALPHA, &secret_key.key_lv1, p),
145          p,
146        )
147      })
148    })
149  })
150}
